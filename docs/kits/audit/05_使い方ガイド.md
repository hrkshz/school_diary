## 🚀 クイックスタート

### Step 1: モデルに履歴記録を追加

```python
# your_app/models.py

from django.db import models
from kits.audit.models import AuditMixin

class Product(AuditMixin):  # ← AuditMixinを継承
    """商品マスター"""
    name = models.CharField(max_length=200)
    price = models.DecimalField(max_digits=10, decimal_places=2)
    stock = models.IntegerField()
    
    # これだけで履歴記録が有効化される！
```

## 📋 基本的な使い方

### パターン1: モデルの変更履歴を記録

```python
from kits.audit.models import AuditMixin

class OvertimeRequest(AuditMixin):
    """残業申請"""
    employee = models.ForeignKey(User, on_delete=models.CASCADE)
    date = models.DateField()
    hours = models.DecimalField(max_digits=4, decimal_places=2)
    status = models.CharField(max_length=20, default='draft')

# ビューで
def submit_overtime(request, overtime_id):
    overtime = OvertimeRequest.objects.get(id=overtime_id)
    
    # 変更前の状態をログに記録
    overtime._history_user = request.user
    overtime._change_reason = "残業申請を提出"
    overtime.status = 'submitted'
    overtime.save()
    
    return redirect('overtime_list')
```

### パターン2: カスタムイベントを記録

```python
from kits.audit.services import AuditService

def approve_overtime(request, overtime_id):
    overtime = OvertimeRequest.objects.get(id=overtime_id)
    service = AuditService()
    
    # 承認処理
    overtime.status = 'approved'
    overtime._history_user = request.user
    overtime._change_reason = "課長が承認"
    overtime.save()
    
    # カスタムイベントログ
    service.log_event(
        event_type="approve",
        event_name="残業申請承認",
        obj=overtime,
        user=request.user,
        description=f"{overtime.employee.username}の残業申請を承認",
        changes={
            "status": {"from": "submitted", "to": "approved"}
        },
        metadata={
            "hours": str(overtime.hours),
            "date": str(overtime.date)
        },
        user_ip=request.META.get('REMOTE_ADDR'),
        user_agent=request.META.get('HTTP_USER_AGENT', '')
    )
    
    return redirect('overtime_list')
```

# ビュー
def overtime_history(request, overtime_id):

## 🔍 高度な使い方

### パターン4: 変更内容を比較

```python
from kits.audit.services import AuditService

def compare_versions(request, overtime_id):
    overtime = OvertimeRequest.objects.get(id=overtime_id)
    service = AuditService()
    
    # 最新2つのバージョンを取得
    history = overtime.history.all()[:2]
    
    if len(history) >= 2:
        # 比較
        diff = service.compare_history(
            overtime,
            history[1].history_id,  # 古いバージョン
            history[0].history_id   # 新しいバージョン
        )
        
        # 変更されたフィールドを表示
        for field, change in diff.items():
            print(f"{field}: {change['old']} → {change['new']}")
```

### パターン6: 監査レポートを生成

```python
from datetime import datetime
from kits.audit.services import AuditService

def monthly_audit_report(request, year, month):
    service = AuditService()
    
    # 月初と月末を計算
    start_date = datetime(year, month, 1)
    if month == 12:
        end_date = datetime(year + 1, 1, 1)
    else:
        end_date = datetime(year, month + 1, 1)
    
    # レポート生成
    report = service.generate_audit_report(
        start_date=start_date,
        end_date=end_date
    )
    
    return render(request, 'audit_report.html', {
        'report': report,
        'year': year,
        'month': month
    })
```

## 🛠️ 実践的なパターン

### kits.approvalsとの連携

```python
from kits.approvals.services import ApprovalService
from kits.audit.services import AuditService

def create_approval_with_audit(request, document_id):
    document = Document.objects.get(id=document_id)
    approval_service = ApprovalService()
    audit_service = AuditService()
    
    # 承認フロー作成
    approval_request = approval_service.create_request(
        workflow=my_workflow,
        content_object=document,
        requester=request.user
    )
    
    # 監査ログ記録
    audit_service.log_event(
        event_type="submit",
        event_name="承認依頼作成",
        obj=document,
        user=request.user,
        description="承認フローを開始しました",
        metadata={
            "workflow": my_workflow.name,
            "approval_request_id": str(approval_request.id)
        }
    )
```

## 📅 定期タスクの設定

### Celery Beatで自動クリーンアップ

```python
# config/settings/base.py

from celery.schedules import crontab

CELERY_BEAT_SCHEDULE = {
    # 毎日深夜2時に古いログをクリーンアップ
    'cleanup-old-audit-logs': {
        'task': 'kits.audit.tasks.cleanup_old_audit_logs',
        'schedule': crontab(hour=2, minute=0),
        'kwargs': {'days': 365}  # 1年以上古いログを削除
    },
    
    # 毎週日曜日に週次レポート生成
    'weekly-audit-report': {
        'task': 'kits.audit.tasks.generate_weekly_audit_report',
        'schedule': crontab(day_of_week=0, hour=8, minute=0)
    },
}
```

## ⚠️ 注意事項

### 1. パフォーマンスへの影響

**大量の更新を行う場合、履歴記録は一時的に無効化**:

```python
from simple_history.utils import update_change_reason

# 一括更新（履歴なし）
Product.objects.filter(category='electronics').update(price=F('price') * 1.1)

# 後から履歴を追加
for product in Product.objects.filter(category='electronics'):
    update_change_reason(product, "一括価格改定")
```

### 2. 個人情報の取り扱い

個人情報を含む履歴は**適切な保持期間**を設定:

```python
# 3年後に自動削除
cleanup_old_audit_logs.delay(days=1095)
```

### 3. ストレージ容量

履歴は積み重なるため、定期的にアーカイブ:

```python
# 90日以上古いログをCSV形式でアーカイブ
from kits.audit.tasks import archive_old_audit_logs

archive_old_audit_logs.delay(days=90)
```
