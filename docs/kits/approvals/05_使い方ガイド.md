## 🚀 基本的な使い方

### ステップ1: グループ（承認者ロール）を作成

```python
from django.contrib.auth.models import Group

# 承認者のグループを作成
manager_group = Group.objects.create(name="Manager")  # 課長
director_group = Group.objects.create(name="Director")  # 部長
```

### ステップ2: 承認ワークフローを作成

```python
from kits.approvals.models import ApprovalWorkflow

# 2段階承認フローを作成
workflow = ApprovalWorkflow.objects.create(
    name="2段階承認（課長→部長）",
    description="課長による一次承認、部長による最終承認",
    is_active=True,
    default_deadline_hours=48,  # デフォルト期限: 48時間
)
```

### ステップ3: 承認ステップを定義

```python
# Step 1: 課長承認
workflow.steps.create(
    order=1,
    name="課長承認",
    description="課長による一次承認",
    approver_role=manager_group,
)

# Step 2: 部長承認
workflow.steps.create(
    order=2,
    name="部長承認",
    description="部長による最終承認",
    approver_role=director_group,
)
```

### ステップ4: 承認依頼を作成・提出

```python
from kits.approvals.services import ApprovalService

service = ApprovalService()

# 承認依頼を作成（例: 残業申請オブジェクト）
request = service.create_request(
    workflow=workflow,
    content_object=overtime_request,  # 任意のモデルオブジェクト
    requester=employee,
    metadata={"reason": "プロジェクトの納期対応"},
)

# 承認依頼を提出
request = service.submit_request(request)
# → status: "pending", current_step: Step 1
```

### ステップ5: 承認する

```python
# 課長が第1ステップを承認
request = service.approve_step(
    request=request,
    step=request.current_step,
    approver=manager,
    comment="承認します。",
)
# → current_step: Step 2に進む

# 部長が第2ステップを承認
request = service.approve_step(
    request=request,
    step=request.current_step,
    approver=director,
    comment="承認します。",
)
# → status: "approved", current_step: None（完了）
```

## 📋 実践例

### 例1: 残業申請の承認フロー

```python
from django.contrib.auth import get_user_model
from django.contrib.auth.models import Group
from kits.approvals.models import ApprovalWorkflow
from kits.approvals.services import ApprovalService

User = get_user_model()

# 1. ユーザーとグループを準備
employee = User.objects.get(email="employee@example.com")
manager = User.objects.get(email="manager@example.com")
director = User.objects.get(email="director@example.com")

manager_group = Group.objects.get(name="Manager")
manager.groups.add(manager_group)

director_group = Group.objects.get(name="Director")
director.groups.add(director_group)

# 2. 残業申請ワークフローを作成
overtime_workflow = ApprovalWorkflow.objects.create(
    name="残業申請承認フロー",
    default_deadline_hours=24,
)

overtime_workflow.steps.create(
    order=1,
    name="上長承認",
    approver_role=manager_group,
)

overtime_workflow.steps.create(
    order=2,
    name="部長承認",
    approver_role=director_group,
)

# 3. 残業申請を作成（仮のモデル）
from myapp.models import OvertimeRequest

overtime = OvertimeRequest.objects.create(
    employee=employee,
    date="2025-10-06",
    hours=3,
    reason="プロジェクトの納期対応",
)

# 4. 承認依頼を作成・提出
service = ApprovalService()
approval = service.create_request(
    workflow=overtime_workflow,
    content_object=overtime,
    requester=employee,
)
approval = service.submit_request(approval)

print(f"Status: {approval.status}")  # → "pending"
print(f"Current step: {approval.current_step}")  # → "上長承認"

# 5. 上長が承認
approval = service.approve_step(
    request=approval,
    step=approval.current_step,
    approver=manager,
    comment="承認します。",
)

print(f"Current step: {approval.current_step}")  # → "部長承認"

# 6. 部長が承認
approval = service.approve_step(
    request=approval,
    step=approval.current_step,
    approver=director,
    comment="承認します。",
)

print(f"Status: {approval.status}")  # → "approved"
print(f"Completed at: {approval.completed_at}")  # → 完了日時
```

### 例2: 並列承認（3人中2人の承認が必要）

```python
# 1. レビュアーグループを作成
reviewer_group = Group.objects.create(name="Reviewer")

reviewer1 = User.objects.get(email="reviewer1@example.com")
reviewer2 = User.objects.get(email="reviewer2@example.com")
reviewer3 = User.objects.get(email="reviewer3@example.com")

reviewer1.groups.add(reviewer_group)
reviewer2.groups.add(reviewer_group)
reviewer3.groups.add(reviewer_group)

# 2. 並列承認ワークフローを作成
parallel_workflow = ApprovalWorkflow.objects.create(
    name="レビュー承認フロー（3人中2人）",
    default_deadline_hours=72,
)

parallel_workflow.steps.create(
    order=1,
    name="レビュアー承認",
    approver_role=reviewer_group,
    is_parallel=True,  # 並列承認を有効化
    required_approvals=2,  # 3人中2人の承認が必要
)

# 3. レビュー依頼を作成・提出
review_request = service.create_request(
    workflow=parallel_workflow,
    content_object=document,
    requester=author,
)
review_request = service.submit_request(review_request)

# 4. レビュアー1が承認
review_request = service.approve_step(
    request=review_request,
    step=review_request.current_step,
    approver=reviewer1,
    comment="問題ありません。",
)

print(f"Status: {review_request.status}")  # → "pending"（まだ1人のみ）

# 5. レビュアー2が承認
review_request = service.approve_step(
    request=review_request,
    step=review_request.current_step,
    approver=reviewer2,
    comment="承認します。",
)

print(f"Status: {review_request.status}")  # → "approved"（2人承認で完了）
```

### 例3: 自動承認（申請者が承認者ロールに属している場合）

```python
# 1. 自動承認ワークフローを作成
auto_workflow = ApprovalWorkflow.objects.create(
    name="自動承認付き承認フロー",
    default_deadline_hours=48,
)

# Step 1: 課長承認（申請者が課長なら自動承認）
auto_workflow.steps.create(
    order=1,
    name="課長承認",
    approver_role=manager_group,
    auto_approve_if_requester_in_role=True,  # 自動承認を有効化
)

# Step 2: 部長承認
auto_workflow.steps.create(
    order=2,
    name="部長承認",
    approver_role=director_group,
)

# 2. 課長が申請（課長承認ステップは自動で通過）
manager_request = service.create_request(
    workflow=auto_workflow,
    content_object=document,
    requester=manager,  # 申請者が課長
)
manager_request = service.submit_request(manager_request)

print(f"Current step: {manager_request.current_step}")
# → "部長承認"（課長承認は自動で通過）
```

### 例4: 承認の否認

```python
# 承認依頼を作成・提出
request = service.create_request(
    workflow=workflow,
    content_object=document,
    requester=employee,
)
request = service.submit_request(request)

# 課長が否認
request = service.reject_step(
    request=request,
    step=request.current_step,
    approver=manager,
    comment="要件を満たしていないため、否認します。",
)

print(f"Status: {request.status}")  # → "rejected"
print(f"Completed at: {request.completed_at}")  # → 否認日時
```

### 例5: 申請者への差戻し

```python
# 承認依頼を作成・提出
request = service.create_request(
    workflow=workflow,
    content_object=document,
    requester=employee,
)
request = service.submit_request(request)

# 課長が差戻し
request = service.return_to_requester(
    request=request,
    step=request.current_step,
    approver=manager,
    comment="記載内容に不足があるため、修正して再提出してください。",
)

print(f"Status: {request.status}")  # → "draft"（下書きに戻る）
print(f"Current step: {request.current_step}")  # → None

# 申請者が修正して再提出
request = service.submit_request(request)  # 再提出
```

### 例8: Celeryタスクで定期チェック

```python
# config/settings/base.py に追加

from celery.schedules import crontab

CELERY_BEAT_SCHEDULE = {
    # 1時間ごとに期限切れをチェック
    "check-overdue-approvals": {
        "task": "kits.approvals.check_overdue_requests",
        "schedule": crontab(minute=0),  # 毎時0分に実行
    },
    # 毎日9時にリマインダーを送信（24時間前）
    "send-approval-reminders": {
        "task": "kits.approvals.send_reminder_notifications",
        "schedule": crontab(hour=9, minute=0),
        "kwargs": {"hours_before_deadline": 24},
    },
}
```

### 例9: 承認履歴の取得

```python
from kits.approvals.models import ApprovalAction

# 特定の承認依頼の全アクション履歴を取得
actions = ApprovalAction.objects.filter(request=request).order_by("created_at")

for action in actions:
    print(f"{action.created_at}: {action.get_action_display()} by {action.approver}")
    print(f"  Step: {action.step.name}")
    print(f"  Comment: {action.comment}")

# django-simple-historyの履歴も参照可能
history = request.history.all()
for record in history:
    print(f"{record.history_date}: {record.history_change_reason}")
    print(f"  Changed by: {record.history_user}")
```

## 🔧 Django管理画面での操作

### 承認ワークフローを管理画面で作成

1. `/admin/approvals/approvalworkflow/add/` にアクセス
2. ワークフロー名、デフォルト期限を入力
3. 「保存して承認ステップを追加」をクリック
4. 承認ステップを追加（order, name, approver_roleを指定）
5. 保存

### 承認依頼を管理画面で確認

1. `/admin/approvals/approvalrequest/` にアクセス
2. ステータス、期限、承認者でフィルタリング
3. 個別の承認依頼をクリックして詳細を確認
4. 承認アクション履歴がインラインで表示される

## 💡 よくある使い方のヒント

### ヒント1: 承認完了後のアクション

```python
# 承認完了をチェックして、次のアクションを実行
if approval.status == "approved":
    # 例: 経費申請が承認されたら、支払い処理を開始
    expense.process_payment()
```

### ヒント2: 通知との連携

kits.notificationsが利用可能な場合、承認依頼・完了・リマインダーが自動で送信されます。

通知テンプレートを作成：

```python
from kits.notifications.models import NotificationTemplate

# 承認依頼の通知テンプレート
NotificationTemplate.objects.create(
    code="approval_request_pending",
    name="承認依頼",
    subject="【承認依頼】{{workflow_name}}",
    body="""
承認依頼が届いています。

ワークフロー: {{workflow_name}}
ステップ: {{step_name}}
申請者: {{requester_name}}

承認をお願いいたします。
    """,
)
```

### ヒント3: カスタム権限チェック

```python
# 承認前に追加のチェックを実行
def custom_approve(request, step, approver):
    # 独自のビジネスロジック
    if not check_custom_permission(approver):
        raise PermissionError("承認権限がありません")

    # 承認を実行
    return service.approve_step(request, step, approver)
```
