## 🎯 基本的な使い方

### 例1: シンプルなCSVレポート生成

**ユースケース**: ユーザー一覧をCSVでエクスポート

#### ステップ1: レポートテンプレートを作成

```python
from kits.reports.models import ReportTemplate, ReportFormat

# テンプレート作成（既に存在する場合はスキップ）
template, created = ReportTemplate.objects.get_or_create(
    code='user_list',
    defaults={
        'name': 'ユーザー一覧',
        'description': 'アクティブなユーザーの一覧をエクスポート',
        'model_name': 'school_diary.users.User',  # Djangoモデルパス
        'supported_formats': [ReportFormat.CSV, ReportFormat.XLSX],
        'default_format': ReportFormat.CSV,
        'is_active': True,
    }
)

if created:
    print("テンプレート作成しました！")
else:
    print("テンプレートは既に存在します")
```

#### ステップ2: レポート生成

```python
from django.contrib.auth import get_user_model
from kits.reports.services import ReportService

User = get_user_model()

# サービスインスタンス作成
service = ReportService()

# 現在のユーザーを取得
user = User.objects.first()

# レポート生成
report = service.generate_report(
    template=template,
    user=user,
    output_format=ReportFormat.CSV,
)

# 結果確認
print(f"レポート生成完了: {report.title}")
print(f"ファイル: {report.file.url}")
print(f"行数: {report.row_count}")
print(f"ファイルサイズ: {report.file_size / 1024:.2f} KB")
```

#### ステップ3: ダウンロード

```python
# ダウンロードURLを取得
download_url = report.file.url
print(f"ダウンロード: http://localhost:8000{download_url}")

# ダウンロード回数を記録
report.increment_download_count()
print(f"ダウンロード回数: {report.download_count}")
```

### 例2: Chart.jsグラフ生成

**ユースケース**: 月次売上データを折れ線グラフで可視化

```python
import pandas as pd
from kits.reports.charts import ChartBuilder

# サンプルデータ
data = {
    '月': ['1月', '2月', '3月', '4月', '5月', '6月'],
    '売上': [120, 190, 300, 250, 200, 310],
    '費用': [80, 100, 150, 120, 90, 180],
}
df = pd.DataFrame(data)

# 折れ線グラフを生成
chart = ChartBuilder.line(
    labels=df['月'].tolist(),
    datasets=[
        {
            'label': '売上',
            'data': df['売上'].tolist(),
            'borderColor': '#4F46E5',
            'backgroundColor': 'rgba(79, 70, 229, 0.1)',
        },
        {
            'label': '費用',
            'data': df['費用'].tolist(),
            'borderColor': '#EF4444',
            'backgroundColor': 'rgba(239, 68, 68, 0.1)',
        },
    ],
    title='月次売上・費用推移',
    y_axis_label='金額（万円）',
)

# JSON出力（Djangoテンプレートに渡す）
chart_json = chart.to_json()
print("Chart.js設定:")
print(chart_json)
```

#### Djangoビューでの使用

```python
# views.py
from django.shortcuts import render
from kits.reports.charts import ChartBuilder

def sales_dashboard(request):
    # データ取得（実際のモデルから）
    # sales_data = SalesRecord.objects.filter(...)

    # グラフ生成
    chart = ChartBuilder.line(
        labels=['1月', '2月', '3月', '4月', '5月', '6月'],
        datasets=[
            {'label': '売上', 'data': [120, 190, 300, 250, 200, 310]}
        ],
        title='月次売上推移',
    )

    context = {
        'chart_json': chart.to_json(),
    }
    return render(request, 'sales/dashboard.html', context)
```

### 例3: PDFレポート生成

**ユースケース**: 月次レポートをPDF形式で出力し、印刷・配布

```python
from kits.reports.models import ReportTemplate, ReportFormat
from kits.reports.services import ReportService

# グラフ付きテンプレートを作成
template, created = ReportTemplate.objects.get_or_create(
    code='monthly_sales_report',
    defaults={
        'name': '月次売上レポート',
        'description': 'グラフ付きの月次売上レポート（PDF）',
        'model_name': 'demos.SalesRecord',
        'supported_formats': [ReportFormat.PDF],
        'default_format': ReportFormat.PDF,
        'html_template': 'reports/base_report.html',
        'chart_config': {
            'type': 'line',
            'x_column': 'month',
            'y_columns': ['amount'],
            'title': '月次売上推移',
            'y_axis_label': '売上（円）',
        },
    }
)

# レポート生成
service = ReportService()
user = request.user  # 現在のユーザー

report = service.generate_report(
    template=template,
    user=user,
    output_format=ReportFormat.PDF,
    parameters={
        'date_from': '2025-01-01',
        'date_to': '2025-12-31',
        'title': '2025年 月次売上レポート',
    }
)

print(f"PDFレポート生成完了: {report.file.url}")
```

```python
from django.shortcuts import render

    }
```

### 例6: スケジュール実行（週次レポート）

**ユースケース**: 毎週月曜9時に自動でレポートを生成し、メール送信

```python
from kits.reports.models import ReportTemplate, ReportSchedule
from django.contrib.auth import get_user_model

User = get_user_model()

# テンプレート取得
template = ReportTemplate.objects.get(code='weekly_sales_report')

# スケジュール設定
schedule, created = ReportSchedule.objects.get_or_create(
    template=template,
    defaults={
        'name': '週次売上レポート自動送信',
        'cron_expression': '0 9 * * 1',  # 毎週月曜9時
        'output_format': 'pdf',
        'is_active': True,
        'send_to_emails': ['manager@example.com', 'ceo@example.com'],
    }
)

# 特定ユーザーにも送信
manager = User.objects.get(username='manager')
schedule.send_to_users.add(manager)

print("スケジュール設定完了: 毎週月曜9時に自動送信")
```

#### スケジュールの仕組み

```python
# Celery Beatが自動的に以下を実行：
# - cron_expressionに従ってタスクをスケジュール
# - ReportService.generate_report() を実行
# - send_to_usersとsend_to_emailsに通知・送信
```

### 例12: 有効期限管理

**ユースケース**: 古いレポートを自動削除してストレージ節約

```python
from datetime import timedelta
from django.utils import timezone
from kits.reports.models import Report

# 有効期限切れレポートを確認
expired_reports = Report.objects.filter(
    expires_at__lt=timezone.now(),
    status='completed'
)

print(f"有効期限切れレポート: {expired_reports.count()}件")

# 手動削除
for report in expired_reports:
    print(f"削除: {report.title} ({report.expires_at})")
    report.file.delete()  # ファイル削除
    report.delete()       # レコード削除

# または、Celeryタスクで自動削除
# cleanup_expired_reports.delay()
```

## 💡 ベストプラクティス

### 1. テンプレートコードの命名規則

```python
# ✅ 良い例: 意味が分かる命名
code='monthly_sales_report'
code='employee_overtime_list'
code='weekly_attendance_summary'

# ❌ 悪い例: 意味不明な命名
code='report1'
code='template_a'
code='test'
```

### 2. パラメータの設計

```python
# ✅ 良い例: 明確なパラメータ名
parameters={
    'date_from': '2025-01-01',
    'date_to': '2025-12-31',
    'department': '営業部',
    'include_inactive': False,
}

# ❌ 悪い例: 不明確なパラメータ名
parameters={
    'from': '2025-01-01',
    'd': '営業部',
    'flag': False,
}
```

### 3. 大量データの処理

```python
# ✅ 良い例: queryset.iterator()でメモリ節約
queryset = SalesRecord.objects.filter(year=2025)
report = service.generate_report(
    template=template,
    user=user,
    queryset=queryset.iterator(chunk_size=1000),  # チャンク処理
)

# ❌ 悪い例: 全件読み込み（メモリ枯渇リスク）
queryset = SalesRecord.objects.all()  # 100万件😓
```

### 4. エラーハンドリング

```python
# ✅ 良い例: 例外を処理
from kits.reports.models import ReportTemplate
from kits.reports.services import ReportService

try:
    template = ReportTemplate.objects.get(code='monthly_sales')
    report = service.generate_report(
        template=template,
        user=request.user,
        output_format=ReportFormat.PDF,
    )
    messages.success(request, "レポート生成完了")
except ReportTemplate.DoesNotExist:
    messages.error(request, "テンプレートが見つかりません")
except Exception as e:
    messages.error(request, f"レポート生成エラー: {str(e)}")
    logger.exception("Report generation failed", exc_info=e)
```

### 5. 有効期限の設定

```python
# ✅ 良い例: 用途に応じて有効期限を設定
from datetime import timedelta
from django.utils import timezone

# 一時的なレポート（7日間）
report.expires_at = timezone.now() + timedelta(days=7)

# 長期保存レポート（1年間）
report.expires_at = timezone.now() + timedelta(days=365)

# 無期限（Noneにする）
report.expires_at = None
```

## 📝 まとめ

### 基本的な流れ

1. **テンプレート作成** - 管理画面またはコード
2. **レポート生成** - `service.generate_report()`
3. **ダウンロード** - `report.file.url`

### よく使うコード

```python
# レポート生成の基本形
from kits.reports.services import ReportService
from kits.reports.models import ReportFormat

service = ReportService()
report = service.generate_report(
    template=template,
    user=request.user,
    output_format=ReportFormat.PDF,
    parameters={'key': 'value'}
)
```

### グラフ生成の基本形

```python
from kits.reports.charts import ChartBuilder

chart = ChartBuilder.line(
    labels=['A', 'B', 'C'],
    datasets=[{'label': 'データ', 'data': [1, 2, 3]}],
    title='タイトル',
)

```
